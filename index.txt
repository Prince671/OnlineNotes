  <script>
    let fileToDelete = null;

    // Function to play a sound
    function playDeleteSound() {
      const audio = new Audio('/path/to/delete-sound.mp3');
      audio.play();
    }

    // Show toast notification
    function showToast(message) {
      const toast = document.createElement("div");
      toast.classList.add("toast", "bg-green-500", "text-white", "px-4", "py-2", "rounded-lg", "absolute", "top-5", "left-1/2", "transform", "-translate-x-1/2", "text-center", "animate-fadeInToast");
      toast.textContent = message;

      document.body.appendChild(toast);

      setTimeout(() => {
        toast.classList.add("opacity-0");
        setTimeout(() => {
          toast.remove();
        }, 300);
      }, 3000); // Remove after 3 seconds
    }

    // Open file modal
    function openModal(filename) {
      const modal = document.getElementById('noteModal');
      const modalContent = document.getElementById('modalContent');

      fetch(`/Files/${filename}`)
        .then(response => response.text())
        .then(data => {
          modalContent.innerHTML = `
            <h3 class="text-xl font-bold text-green-400 mb-4">${filename}</h3>
            <pre class="whitespace-pre-wrap break-words text-sm text-zinc-200">${data}</pre>
          `;
        })
        .catch(err => {
          modalContent.innerHTML = `<p class="text-red-500">Error loading file: ${err}</p>`;
        });

      modal.classList.remove('hidden');
      modal.classList.add('animate-fadeInModal');
    }

    // Close file modal
    function closeModal() {
      const modal = document.getElementById('noteModal');
      modal.classList.add('hidden');
    }

    // Confirm delete action
    function confirmDelete(filename) {
      fileToDelete = filename;
      document.getElementById("confirmModal").classList.remove("hidden");
    }

    // Cancel delete
    function cancelDelete() {
      fileToDelete = null;
      document.getElementById("confirmModal").classList.add("hidden");
    }

    // Perform the deletion
    function performDelete() {
      if (!fileToDelete) return;

      const card = document.getElementById(`card-${fileToDelete}`);
      card.classList.remove("animate-flyToBin"); // restart animation
      void card.offsetWidth;
      card.classList.add("animate-flyToBin");

      playDeleteSound(); // Play the delete sound

      setTimeout(() => {
        const form = document.createElement("form");
        form.method = "POST";
        form.action = "/delete";

        const input = document.createElement("input");
        input.type = "hidden";
        input.name = "filename";
        input.value = fileToDelete;

        form.appendChild(input);
        document.body.appendChild(form);
        form.submit();

        showToast("File deleted successfully!"); // Show success toast
      }, 600); // match animation duration

      cancelDelete();
    }
  </script>
